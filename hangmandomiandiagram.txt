object SelectPlayers

Turn *-- WordGiverable
Turn *-- WordGuesserable

WordGuesserable --|> HangManPlayer

WordGuesserable *-- GuessWord

HangedMan *-- SecretWord
HangedMan *-- GuessingAttempts
